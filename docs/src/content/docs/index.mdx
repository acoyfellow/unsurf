```bash
# Scout any site → get a typed OpenAPI spec
curl -X POST https://unsurf-api.coey.dev/tools/scout \
  -H "Content-Type: application/json" \
  -d '{
    "url": "https://any-website.com",
    "task": "find all API endpoints"
  }'

# Returns: siteId, endpoints[], openApiSpec, pathId
```

```typescript
import { scout, worker, heal } from "unsurf";

// Scout → capture every API call
const result = await scout({ url, task });

// Worker → replay without a browser
const data = await worker({ pathId: result.pathId });

// Heal → auto-fix when sites change
const fixed = await heal({ pathId, error });
```
